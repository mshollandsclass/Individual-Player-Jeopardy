    function startGame() {
      const nameInput = document.getElementById('student-name-input').value.trim();
      const numberInput = document.getElementById('student-number-input').value.trim();

        alert(strings[language].noAnswer);
        return;
      }
      studentName = `${nameInput} (${numberInput})`;
      document.getElementById('start-modal').style.display = 'none';

    function renderBoard() {
      const board = document.getElementById('game-board');
      board.style.display = 'grid';
      board.innerHTML = '';
      
      // Update header buttons with current language
      const str = strings[language];
      document.getElementById('round1-btn').innerText = str.round1;
      document.getElementById('round2-btn').innerText = str.round2;
      document.getElementById('final-jeopardy-btn').innerText = str.finalJeopardy;
      document.querySelector('button[onclick="submitFinalAnswers()"]').innerText = str.submitAnswers;
      
      // Update score label
      const scoreSpan = document.querySelector('span[style*="color:#facc15;font-size:26px"]');
      if (scoreSpan) scoreSpan.innerText = str.score;
      
      const data = currentRound === 1 ? (language === 'es' ? round1Questions : round1QuestionsEN) : (language === 'es' ? round2Questions : round2QuestionsEN);
      const names = currentRound === 1 ? (language === 'es' ? catNames.round1 : catNamesEN.round1) : (language === 'es' ? catNames.round2 : catNamesEN.round2);
      const keys = Object.keys(data);

      names.forEach(n => {
        const div = document.createElement('div');
        div.className = 'category-font';
        div.style = "background:#1e3a8a; display:flex; align-items:center; justify-content:center; text-align:center; padding:10px; border-radius:8px; color:#facc15; font-size:24px; border:2px solid #eab308";
        div.innerText = n;
        board.appendChild(div);
      });

      for (let i = 0; i < 5; i++) {
        keys.forEach(k => {
          const q = data[k][i];
          const id = `${currentRound}-${k}-${i}`;
          const card = document.createElement('div');
          card.className = 'card money-font' + (used.has(id) ? ' used' : '');
          card.style.fontSize = "48px";
          card.innerText = used.has(id) ? '' : `$${q.value}`;
          if (!used.has(id)) card.onclick = () => openQuestion(q, id, k);
          board.appendChild(card);
        });
      }
    }

    function openQuestion(q, id, category) {
      currentQ = { ...q, id, category };
      const catList = currentRound === 1 ? (language === 'es' ? catNames.round1 : catNamesEN.round1) : (language === 'es' ? catNames.round2 : catNamesEN.round2);
      const keys = Object.keys(currentRound === 1 ? (language === 'es' ? round1Questions : round1QuestionsEN) : (language === 'es' ? round2Questions : round2QuestionsEN));
      currentCategory = catList[keys.indexOf(category)];
      
      const str = strings[language];
      document.getElementById('modal-value').innerText = `$${q.value}`;
      document.getElementById('modal-question').innerText = q.q;
      document.getElementById('modal-answer').innerText = q.a;
      document.getElementById('student-answer-input').placeholder = str.enterAnswer;
      document.getElementById('student-answer-input').value = '';
      
      // Update button text
      const submitBtn = document.querySelector('button[onclick="submitAnswer()"]');
      if (submitBtn) submitBtn.innerText = str.submit;
      
      const correctAnswerLabel = document.querySelector('div#modal p[style*="color:#facc15;font-size:32px"]');
      if (correctAnswerLabel) correctAnswerLabel.innerText = str.correctAnswer;
      
      document.getElementById('answer-section').style.display = 'block';
      document.getElementById('result-section').style.display = 'none';
      document.getElementById('modal').style.display = 'flex';
    }

    async function submitAnswer() {
      const studentAnswer = document.getElementById('student-answer-input').value.trim();
      const str = strings[language];
      
      if (!studentAnswer) {
        alert(str.noAnswer);
        return;
      }

      const correctAnswer = currentQ.a.replace(/What is |What are |\?|¿Qué es |¿Qué son /g, '');
      const isCorrect = studentAnswer.toLowerCase().includes(correctAnswer.toLowerCase()) || correctAnswer.toLowerCase().includes(studentAnswer.toLowerCase());
      
      const pointsEarned = isCorrect ? currentQ.value : 0;
      if (isCorrect) {
        studentScore += pointsEarned;
      }

      // Prepare response data
      const responseData = {
        student_name: studentName,
        round: String(currentRound),
        category: currentCategory,
        question_value: String(currentQ.value),
        student_answer: studentAnswer,
        correct_answer: currentQ.a,
        is_correct: String(isCorrect),
        points_earned: String(pointsEarned),
        final_score: String(studentScore),
        timestamp: new Date().toISOString()
      };

      // Save to Data SDK immediately
      if (window.dataSdk) {
        const result = await window.dataSdk.create(responseData);
        if (result.isOk) {
          console.log("Response saved immediately:", responseData);
        } else {

      const str = strings[language];
      
      if (!studentAnswer) {
        alert(str.noAnswer);
        return;
      }

      const isCorrect = studentAnswer.toLowerCase().includes('buzo') || studentAnswer.toLowerCase().includes('diver');
      const pointsEarned = isCorrect ? 1000 : 0;
      
      if (isCorrect) {
        studentScore += pointsEarned;
      }

      const correctAnswerDisplay = language === 'es' ? 'Buzo / Diver' : 'Diver / Buzo';

      // Prepare response data
      const responseData = {
        student_name: studentName,
        round: String(3),
        category: 'FINAL JEOPARDY',
        question_value: String(1000),
        student_answer: studentAnswer,
        correct_answer: correctAnswerDisplay,
        is_correct: String(isCorrect),
        points_earned: String(pointsEarned),
        final_score: String(studentScore),
        timestamp: new Date().toISOString()
      };

      // Save to Data SDK immediately
      if (window.dataSdk) {
        const result = await window.dataSdk.create(responseData);
        if (result.isOk) {
          console.log("FJ Response saved immediately:", responseData);
        } else {
          console.error("Failed to save FJ response:", result.error);
        }
      }

      allResponses.push(responseData);

      document.getElementById('student-score').innerText = `$${studentScore}`;
      document.getElementById('fj-feedback-message').innerText = isCorrect ? str.yourAnswerCorrect : str.incorrect;
